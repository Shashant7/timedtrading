//@version=6
indicator("Price vs Earnings - Advanced (Growth & Forecast)", overlay=true, max_bars_back=5000)

// ==================== INPUTS ====================
// Valuation Mode
peMode = input.string("Auto (Historical Avg)", "P/E Valuation Mode", 
     options=["Auto (Historical Avg)", "Auto (Median)", "Growth-Adjusted PEG", "Manual"], 
     group="Valuation", 
     tooltip="Growth-Adjusted uses PEG ratio to account for earnings growth")
manualPE = input.float(15, "Manual P/E Multiple", minval=5, maxval=100, group="Valuation")
targetPEG = input.float(1.0, "Target PEG Ratio", minval=0.5, maxval=3.0, group="Valuation", 
     tooltip="PEG = 1.0 is considered fair value. Lower = more attractive")
maxGrowthPE = input.float(40, "Maximum P/E from Growth", minval=20, maxval=100, group="Valuation",
     tooltip="Cap the growth-adjusted P/E at this level to avoid unrealistic valuations")

// Growth Calculation
growthLookback = input.int(20, "Growth Lookback (quarters)", minval=4, maxval=40, group="Growth Analysis",
     tooltip="Number of quarters to calculate earnings growth trend")
forecastPeriod = input.int(252, "Forecast Period (bars)", minval=50, maxval=1000, group="Growth Analysis",
     tooltip="How far ahead to project earnings growth")

// Historical Analysis
historicalPeriod = input.int(1260, "Historical P/E Period (bars)", minval=252, maxval=5000, group="Valuation")
earningsSmooth = input.int(4, "Earnings Smoothing (quarters)", minval=1, maxval=20, group="Calculation")

// Display Options
showPrice = input.bool(true, "Show Price", group="Display")
showFairValue = input.bool(true, "Show Fair Value Line", group="Display")
showEPSLine = input.bool(true, "Show Growth-Adjusted Line", group="Display")
showGrowthOnlyWhenDifferent = input.bool(true, "Hide Growth Line When Same as Fair Value", group="Display", 
     tooltip="In Growth-Adjusted PEG mode, blue and teal are identical - this hides the teal line to avoid confusion")
showForecast = input.bool(true, "Show EPS Forecast", group="Display")
showBands = input.bool(true, "Show Valuation Bands", group="Display")
bandWidth = input.float(15, "Valuation Band Width (%)", minval=5, maxval=50, group="Display")
useAdaptiveBands = input.bool(true, "Use Adaptive Bands for Growth Mode", group="Display",
     tooltip="When enabled, uses wider bands (25%) in Growth-Adjusted mode since growth stocks justify higher volatility")

// Colors
colorPrice = input.color(color.new(#FF5252, 0), "Price Color", group="Colors")
colorFairValue = input.color(color.new(#2196F3, 0), "Fair Value Color", group="Colors")
colorEPS = input.color(color.new(#00897B, 0), "EPS Line Color", group="Colors")
colorForecast = input.color(color.new(#9C27B0, 60), "Forecast Color", group="Colors")
colorOvervalued = input.color(color.new(#F44336, 90), "Overvalued Zone", group="Colors")
colorUndervalued = input.color(color.new(#4CAF50, 90), "Undervalued Zone", group="Colors")

// ==================== GET EARNINGS DATA ====================

// Try multiple financial metrics to find available data
epsBasic = request.financial(syminfo.tickerid, "EARNINGS_PER_SHARE_BASIC", "TTM")
epsDiluted = request.financial(syminfo.tickerid, "EARNINGS_PER_SHARE_DILUTED", "TTM")
epsReported = request.financial(syminfo.tickerid, "EARNINGS_PER_SHARE", "TTM")

// Get quarterly data for growth calculation
epsQuarterly = request.financial(syminfo.tickerid, "EARNINGS_PER_SHARE_DILUTED", "FQ")

// Select the best available EPS value
epsRaw = not na(epsDiluted) ? epsDiluted :
         not na(epsBasic) ? epsBasic :
         not na(epsReported) ? epsReported :
         not na(epsQuarterly) ? epsQuarterly * 4 : 
         na

// Check if we have valid data
hasEarningsData = not na(epsRaw) and epsRaw != 0

// Smooth the earnings
epsSmooth = ta.sma(epsRaw, earningsSmooth)

// ==================== CALCULATE EARNINGS GROWTH RATE ====================

// Collect historical EPS values for growth calculation
var float[] epsHistory = array.new_float(0)
var int[] barHistory = array.new_int(0)

if not na(epsSmooth) and epsSmooth != 0
    // Only add if value changed (new quarterly data)
    lastEPS = array.size(epsHistory) > 0 ? array.get(epsHistory, array.size(epsHistory) - 1) : 0.0
    if epsSmooth != lastEPS
        array.push(epsHistory, epsSmooth)
        array.push(barHistory, bar_index)
        
        // Keep array size manageable
        if array.size(epsHistory) > growthLookback
            array.shift(epsHistory)
            array.shift(barHistory)

// Calculate growth rate using linear regression
var float epsGrowthRate = na
var float epsIntercept = na

if array.size(epsHistory) >= 4 // Need at least 4 quarters
    // Prepare data for linear regression
    n = array.size(epsHistory)
    
    // Calculate means
    sumX = 0.0
    sumY = 0.0
    for i = 0 to n - 1
        sumX := sumX + i
        sumY := sumY + array.get(epsHistory, i)
    
    meanX = sumX / n
    meanY = sumY / n
    
    // Calculate slope (growth rate)
    numerator = 0.0
    denominator = 0.0
    for i = 0 to n - 1
        x_diff = i - meanX
        y_diff = array.get(epsHistory, i) - meanY
        numerator := numerator + (x_diff * y_diff)
        denominator := denominator + (x_diff * x_diff)
    
    slope = denominator != 0 ? numerator / denominator : 0.0
    intercept = meanY - (slope * meanX)
    
    // Store for use
    epsIntercept := intercept
    
    // Convert slope to annualized percentage growth rate
    // Slope is per quarter, so annualize it
    avgEPS = meanY
    if avgEPS != 0
        quarterlyGrowth = slope / avgEPS
        epsGrowthRate := quarterlyGrowth * 4 * 100 // Annualized percentage

// ==================== CALCULATE P/E METRICS ====================

// Current P/E ratio
currentPE = hasEarningsData and epsRaw > 0 ? close / epsRaw : na

// Collect historical P/E values
var float[] peHistory = array.new_float(0)

if not na(currentPE) and currentPE > 0 and currentPE < 200
    array.push(peHistory, currentPE)
    if array.size(peHistory) > historicalPeriod
        array.shift(peHistory)

// Calculate historical average and median P/E
historicalAvgPE = array.size(peHistory) > 20 ? array.avg(peHistory) : na
historicalMedianPE = array.size(peHistory) > 20 ? array.median(peHistory) : na

// Calculate PEG ratio (P/E divided by growth rate)
pegRatio = not na(currentPE) and not na(epsGrowthRate) and epsGrowthRate > 0 ? 
           currentPE / epsGrowthRate : na

// Determine fair value P/E based on mode
var float fairValuePE = na

if peMode == "Auto (Historical Avg)"
    fairValuePE := historicalAvgPE
else if peMode == "Auto (Median)"
    fairValuePE := historicalMedianPE
else if peMode == "Growth-Adjusted PEG"
    // Fair P/E = Growth Rate × Target PEG
    // If growth is 20% and target PEG is 1.0, fair P/E = 20
    fairValuePE := not na(epsGrowthRate) and epsGrowthRate > 0 ? epsGrowthRate * targetPEG : historicalAvgPE
else // Manual
    fairValuePE := manualPE

// ==================== CALCULATE FAIR VALUE ====================

// Fair value = EPS × Fair P/E
fairValue = (hasEarningsData and not na(epsSmooth) and not na(fairValuePE) ? 
             epsSmooth * fairValuePE : na)

// Calculate valuation bands with adaptive width
// Growth stocks need wider bands due to higher volatility and uncertainty
effectiveBandWidth = bandWidth
if useAdaptiveBands and peMode == "Growth-Adjusted PEG"
    effectiveBandWidth := bandWidth * 1.67 // 15% → 25%, 20% → 33%

upperBand = not na(fairValue) ? fairValue * (1 + effectiveBandWidth / 100) : na
lowerBand = not na(fairValue) ? fairValue * (1 - effectiveBandWidth / 100) : na

// ==================== THREE VALUATION LINES ====================

// Line 1: Fair Value = Based on selected mode
// (Already calculated above in fairValuePE)

// Line 2: Growth-Adjusted Value = Uses PEG framework with reasonable caps
// This shows what growth justifies, with sensible limits
float growthAdjustedValue = na
var float actualGrowthPE = na

if not na(epsSmooth) and not na(epsGrowthRate) and epsGrowthRate > 0
    // PEG Theory: Fair P/E = Growth Rate when PEG = 1.0
    growthBasedPE = epsGrowthRate * targetPEG
    
    // Apply reasonable caps:
    // - Minimum: Use historical avg if it's higher (don't go below normal valuation)
    // - Maximum: User-defined cap (default 40×)
    minPE = not na(historicalAvgPE) ? historicalAvgPE : 15.0
    
    cappedPE = math.max(minPE, math.min(growthBasedPE, maxGrowthPE))
    actualGrowthPE := cappedPE
    growthAdjustedValue := epsSmooth * cappedPE
else
    // No growth or negative growth - don't show growth-adjusted line
    actualGrowthPE := na
    growthAdjustedValue := na

// Line 3: Conservative Value = Current EPS × (Historical Median × 0.85)
// A more conservative floor - represents "bargain" territory
conservativeValue = (hasEarningsData and not na(epsSmooth) and not na(historicalMedianPE) ? 
                     epsSmooth * (historicalMedianPE * 0.85) : na)

// Add explanation for what each mode shows
var string modeExplanation = ""
if peMode == "Auto (Historical Avg)"
    modeExplanation := "Blue: Historical avg P/E | Teal: Growth-justified P/E"
else if peMode == "Auto (Median)"
    modeExplanation := "Blue: Median P/E | Teal: Growth-justified P/E"
else if peMode == "Growth-Adjusted PEG"
    modeExplanation := "Blue: Growth-justified (same as Teal)"
else
    modeExplanation := "Blue: Manual P/E | Teal: Growth-justified P/E"

// ==================== FORECAST EARNINGS ====================

// Project EPS forward based on growth rate
var line[] forecastLines = array.new_line(0)

// Clear old forecast lines
if barstate.islast and array.size(forecastLines) > 0
    for i = 0 to array.size(forecastLines) - 1
        line.delete(array.get(forecastLines, i))
    array.clear(forecastLines)

// Draw forecast if we have growth data
if barstate.islast and showForecast and not na(epsGrowthRate) and not na(epsSmooth)
    // Starting point
    currentBar = bar_index
    currentEPS = epsSmooth
    currentPrice = currentEPS * (not na(fairValuePE) ? fairValuePE : currentPE)
    
    // Project forward quarter by quarter
    quarterBars = 63 // Approximate bars per quarter (252 trading days / 4)
    numQuarters = math.floor(forecastPeriod / quarterBars)
    
    if numQuarters > 0 and not na(epsGrowthRate)
        quarterlyGrowthRate = epsGrowthRate / 4 / 100 // Convert annual % to quarterly decimal
        
        prevBar = currentBar
        prevEPS = currentEPS
        prevPrice = currentPrice
        
        for i = 1 to math.min(numQuarters, 20) // Limit to 20 quarters (5 years)
            futureBar = currentBar + (i * quarterBars)
            futureEPS = currentEPS * math.pow(1 + quarterlyGrowthRate, i)
            futurePrice = futureEPS * (not na(fairValuePE) ? fairValuePE : currentPE)
            
            // Draw line segment
            forecastLine = line.new(
                 prevBar, prevPrice,
                 futureBar, futurePrice,
                 color=colorForecast,
                 style=line.style_dotted,
                 width=2)
            
            array.push(forecastLines, forecastLine)
            
            prevBar := futureBar
            prevPrice := futurePrice

// ==================== VALUATION METRICS ====================

// Calculate premium/discount to fair value
premium = not na(fairValue) and fairValue != 0 ? ((close - fairValue) / fairValue) * 100 : na

// Determine valuation status using effective band width
isOvervalued = not na(premium) and premium > effectiveBandWidth
isUndervalued = not na(premium) and premium < -effectiveBandWidth

// ==================== PLOTS ====================

plotsEnabled = hasEarningsData and not na(fairValue)

// Plot price
pricePlot = plot(showPrice ? close : na, 
     "Price", 
     color=colorPrice, 
     linewidth=2)

// Plot fair value line
fairValuePlot = plot(showFairValue and plotsEnabled ? fairValue : na, 
     "Fair Value", 
     color=colorFairValue, 
     linewidth=2)

// Plot growth-adjusted value line (accounts for earnings growth rate)
// Hide if in Growth-Adjusted mode and user wants to avoid duplication
showGrowthLine = showEPSLine and plotsEnabled and not na(growthAdjustedValue)
if showGrowthOnlyWhenDifferent and peMode == "Growth-Adjusted PEG"
    showGrowthLine := false // Don't show since it's identical to fair value in this mode

epsGrowthPlot = plot(showGrowthLine ? growthAdjustedValue : na,
     "Growth-Adjusted Value",
     color=colorEPS,
     linewidth=2,
     style=plot.style_line)

// Plot valuation bands
upperBandPlot = plot(showBands and plotsEnabled ? upperBand : na, 
     "Upper Band", 
     color=color.new(color.red, 70), 
     linewidth=1, 
     style=plot.style_circles)

lowerBandPlot = plot(showBands and plotsEnabled ? lowerBand : na, 
     "Lower Band", 
     color=color.new(color.green, 70), 
     linewidth=1, 
     style=plot.style_circles)

// ==================== SHADING ====================

fill(pricePlot, fairValuePlot, 
     color = plotsEnabled and close > fairValue ? colorOvervalued : colorUndervalued,
     title="Valuation Zone")

fill(fairValuePlot, upperBandPlot, color=color.new(color.red, 95), title="Overvalued Band")
fill(fairValuePlot, lowerBandPlot, color=color.new(color.green, 95), title="Undervalued Band")

bgcolor(plotsEnabled and isOvervalued ? color.new(color.red, 97) : 
        plotsEnabled and isUndervalued ? color.new(color.green, 97) : na)

// ==================== INFO TABLE ====================
var table infoTable = table.new(position.top_right, 2, 12, border_width=1)

if barstate.islast and hasEarningsData
    table.clear(infoTable, 0, 0, 1, 11)
    
    // Header
    table.cell(infoTable, 0, 0, "Metric", text_color=color.white, bgcolor=color.gray, text_size=size.small)
    table.cell(infoTable, 1, 0, "Value", text_color=color.white, bgcolor=color.gray, text_size=size.small)
    
    // Current Price
    table.cell(infoTable, 0, 1, "Current Price", text_color=color.white, bgcolor=color.new(color.blue, 50), text_size=size.small)
    table.cell(infoTable, 1, 1, "$" + str.tostring(close, "#.##"), text_color=color.white, bgcolor=color.new(color.blue, 70), text_size=size.small)
    
    // Fair Value
    fairValText = na(fairValue) ? "N/A" : "$" + str.tostring(fairValue, "#.##")
    table.cell(infoTable, 0, 2, "Fair Value", text_color=color.white, bgcolor=color.new(color.teal, 50), text_size=size.small)
    table.cell(infoTable, 1, 2, fairValText, text_color=color.white, bgcolor=color.new(color.teal, 70), text_size=size.small)
    
    // Current P/E
    peText = na(currentPE) ? "N/A" : str.tostring(currentPE, "#.##")
    table.cell(infoTable, 0, 3, "Current P/E", text_color=color.white, bgcolor=color.new(color.purple, 50), text_size=size.small)
    table.cell(infoTable, 1, 3, peText, text_color=color.white, bgcolor=color.new(color.purple, 70), text_size=size.small)
    
    // Fair P/E
    fairPEText = na(fairValuePE) ? "Calculating..." : str.tostring(fairValuePE, "#.##")
    table.cell(infoTable, 0, 4, "Fair P/E", text_color=color.white, bgcolor=color.new(color.purple, 50), text_size=size.small)
    table.cell(infoTable, 1, 4, fairPEText, text_color=color.white, bgcolor=color.new(color.purple, 70), text_size=size.small)
    
    // EPS
    epsText = na(epsRaw) ? "N/A" : str.tostring(epsRaw, "#.##")
    table.cell(infoTable, 0, 5, "EPS (TTM)", text_color=color.white, bgcolor=color.new(color.orange, 50), text_size=size.small)
    table.cell(infoTable, 1, 5, "$" + epsText, text_color=color.white, bgcolor=color.new(color.orange, 70), text_size=size.small)
    
    // EPS Growth Rate
    growthText = na(epsGrowthRate) ? "Calculating..." : str.tostring(epsGrowthRate, "#.#") + "%"
    growthColor = na(epsGrowthRate) ? color.gray : epsGrowthRate > 10 ? color.green : epsGrowthRate > 0 ? color.orange : color.red
    table.cell(infoTable, 0, 6, "EPS Growth (Annual)", text_color=color.white, bgcolor=color.new(color.green, 50), text_size=size.small)
    table.cell(infoTable, 1, 6, growthText, text_color=color.white, bgcolor=color.new(growthColor, 50), text_size=size.normal)
    
    // PEG Ratio
    pegText = na(pegRatio) ? "N/A" : str.tostring(pegRatio, "#.##")
    pegColor = na(pegRatio) ? color.gray : pegRatio < 1 ? color.green : pegRatio < 2 ? color.orange : color.red
    table.cell(infoTable, 0, 7, "PEG Ratio", text_color=color.white, bgcolor=color.new(color.navy, 50), text_size=size.small)
    table.cell(infoTable, 1, 7, pegText, text_color=color.white, bgcolor=color.new(pegColor, 50), text_size=size.normal)
    
    // Growth-Justified P/E (what the teal line uses)
    growthPEText = na(actualGrowthPE) ? "N/A" : str.tostring(actualGrowthPE, "#.##")
    table.cell(infoTable, 0, 8, "Growth-Justified P/E", text_color=color.white, bgcolor=color.new(color.teal, 50), text_size=size.small)
    table.cell(infoTable, 1, 8, growthPEText, text_color=color.white, bgcolor=color.new(color.teal, 70), text_size=size.small)
    
    // Premium/Discount
    premiumText = na(premium) ? "N/A" : str.tostring(premium, "#.#") + "%"
    premiumColor = na(premium) ? color.gray : premium > 0 ? color.red : color.green
    table.cell(infoTable, 0, 9, "Premium/Discount", text_color=color.white, bgcolor=color.new(color.gray, 50), text_size=size.small)
    table.cell(infoTable, 1, 9, premiumText, text_color=color.white, bgcolor=color.new(premiumColor, 50), text_size=size.normal)
    
    // Valuation Mode
    table.cell(infoTable, 0, 10, "Valuation Mode", text_color=color.white, bgcolor=color.new(color.gray, 50), text_size=size.small)
    table.cell(infoTable, 1, 10, peMode, text_color=color.white, bgcolor=color.new(color.gray, 70), text_size=size.small)
    
    // Status
    statusText = isOvervalued ? "OVERVALUED" : isUndervalued ? "UNDERVALUED" : "FAIR"
    statusColor = isOvervalued ? color.red : isUndervalued ? color.green : color.orange
    table.cell(infoTable, 0, 11, "Status", text_color=color.white, bgcolor=color.new(color.gray, 50), text_size=size.small)
    table.cell(infoTable, 1, 11, statusText, text_color=color.white, bgcolor=color.new(statusColor, 30), text_size=size.small)

// Show error if no data
if barstate.islast and not hasEarningsData
    var label warningLabel = na
    label.delete(warningLabel)
    
    warningLabel := label.new(
         bar_index, 
         high * 1.05, 
         "⚠️ NO EARNINGS DATA\n\nTry:\n• Use Manual P/E mode\n• Use ETFs (SPY not SPX)\n• Check different ticker", 
         style=label.style_label_down,
         color=color.new(color.red, 10),
         textcolor=color.white,
         size=size.normal,
         textalign=text.align_left)

// ==================== ALERTS ====================

crossedOverUpperBand = ta.crossover(close, upperBand)
crossedUnderLowerBand = ta.crossunder(close, lowerBand)
crossedFairValue = ta.cross(close, fairValue)

alertcondition(plotsEnabled and crossedOverUpperBand, 
     title="Entered Overvalued Zone",
     message="{{ticker}} price entered overvalued zone")

alertcondition(plotsEnabled and crossedUnderLowerBand,
     title="Entered Undervalued Zone", 
     message="{{ticker}} price entered undervalued zone")

alertcondition(plotsEnabled and crossedFairValue,
     title="Price Crossed Fair Value",
     message="{{ticker}} price crossed fair value line")
